	//Calcular el Indice de Masa Corporal (IMC) según el peso en kg y altura en m del paciente.
	escribir "Ingrese su peso kg:"
	leer p
	escribir "Ingrese su altura en m"
	leer a
	IMC <- P/(a^2)
	escribir "Su IMC es:" IMC " kg/m^2"
	Si IMC <15 entonces
		escribir "Requiere ingreso urgente al hospital"
	FinSi
	Si IMC>=15 Y IMC <=20 Entonces
		escribir "Usted posee bajo peso"
	FinSi
	Si IMC>20 Y IMC<=30 Entonces
		escribir "Usted posee un peso normal (saludable)"
	FinSi
	Si IMC > 30 Y IMC <=40 Entonces
		escribir " Usted posee sobrepeso (obesidad de grado I)"
	FinSi
	Si IMC >40 Entonces
		escribir "Usted posee sobrepeso crónico (obesidad de grado II)"
	FinSi
